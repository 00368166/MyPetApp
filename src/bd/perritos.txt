create table dogos (
  id bigint generated by default as identity primary key,
  user_id uuid references auth.users not null,
  nombre text check (char_length(nombre) > 2),
  tipo text check (char_length(tipo) > 3),
  cumple date,
  descrip text check (char_length(descrip) > 2),
  
  descrip_casa text check (char_length(descrip_casa) > 2),
  
  descrip_direc text check (char_length(descrip_direc) > 2),
  
  descrip_salud text check (char_length(descrip_salud) > 2),
  esterilizado boolean default false,
  inserted_at timestamp with time zone default timezone('utc'::text, now()) not null
);

alter table dogos enable row level security;

create policy "Individuals can create dogos." on dogos for
    insert with check (auth.uid() = user_id);

create policy "Individuals can view their own dogos. " on dogos for
    select using (auth.uid() = user_id);

create policy "Individuals can update their own dogos." on dogos for
    update using (auth.uid() = user_id);

create policy "Individuals can delete their own dogos." on dogos for
    delete using (auth.uid() = user_id);
